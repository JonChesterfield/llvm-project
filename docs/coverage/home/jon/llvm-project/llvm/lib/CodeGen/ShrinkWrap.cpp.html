14
// it 1) dominates (resp. post-dominates) all the frame related operations and

// between 2) two executions of the Save (resp. Restore) point there is an

// execution of the Restore (resp. Save) point.

// for (int i = 0; i < 10; ++i) {

//   Save
21
//   ...

//   Restore

// }

//   Save

//   ...

//   Restore
